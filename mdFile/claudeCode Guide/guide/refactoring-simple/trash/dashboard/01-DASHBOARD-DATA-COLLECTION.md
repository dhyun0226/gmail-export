class DataValidator {
    validate(data) {
        const required = [
            'metrics.before.complexity',
            'metrics.after.complexity',
            'issues.identified',
            'issues.resolved',
            'performance.before',
            'performance.after'

                missing.push(path);

        });

            throw new Error(`Missing required data: ${missing.join(', ')}`);
        }
        
        return true;
    }
    
    getNestedValue(obj, path) {
